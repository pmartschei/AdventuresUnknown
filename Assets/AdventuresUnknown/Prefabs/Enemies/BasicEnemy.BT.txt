tree("Root")
	sequence
		FindNearestTarget()
		fallback
			tree("Attack")
			tree("MoveToTarget")

tree("MoveToTarget")
		sequence
			HasTarget
			not IsNearTarget()
			AimTowardsTarget()
			MoveTowardsTarget()
				
tree("Attack")
	sequence
		HasTarget
		IsNearTarget()
		MarkTarget()
		SetAnimTrigger("Attack")
		WaitWithModifier(0.25)
		Attack()
		WaitWithModifier(1.0)
